=====================================================
DATABASE SCHEMA EXAMPLE (PostgreSQL)
=====================================================

CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    full_name VARCHAR(100) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(20) NOT NULL CHECK (role IN ('ADMIN', 'AUTHOR', 'READER')),
    status VARCHAR(20) NOT NULL DEFAULT 'ACTIVE' CHECK (status IN ('ACTIVE', 'BLOCKED')),
    bio TEXT,
    avatar_url VARCHAR(500),
    articles_count INTEGER DEFAULT 0,
    total_views INTEGER DEFAULT 0,
    join_date DATE NOT NULL DEFAULT CURRENT_DATE,
    last_login TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP WITH TIME ZONE
);

CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_role ON users(role);
CREATE INDEX idx_users_status ON users(status);

=====================================================
PRISMA SCHEMA EXAMPLE
=====================================================

model User {
  id            String    @id @default(uuid())
  fullName      String    @map("full_name") @db.VarChar(100)
  email         String    @unique @db.VarChar(255)
  passwordHash  String    @map("password_hash") @db.VarChar(255)
  role          UserRole
  status        UserStatus @default(ACTIVE)
  bio           String?   @db.Text
  avatarUrl     String?   @map("avatar_url") @db.VarChar(500)
  articlesCount Int       @default(0) @map("articles_count")
  totalViews    Int       @default(0) @map("total_views")
  joinDate      DateTime  @default(now()) @map("join_date") @db.Date
  lastLogin     DateTime? @map("last_login")
  createdAt     DateTime  @default(now()) @map("created_at")
  updatedAt     DateTime  @updatedAt @map("updated_at")
  deletedAt     DateTime? @map("deleted_at")

  @@index([email])
  @@index([role])
  @@index([status])
  @@map("users")
}

enum UserRole {
  ADMIN
  AUTHOR
  READER
}

enum UserStatus {
  ACTIVE
  BLOCKED
}

=====================================================
TESTING SCENARIOS
=====================================================

1. Create User:
   - Valid data → Success
   - Duplicate email → Error 400
   - Invalid email format → Error 400
   - Missing required fields → Error 400
   - Password too short → Error 400

2. Get Users:
   - No filters → Returns all users
   - With search → Returns filtered users
   - With role filter → Returns users with that role
   - With status filter → Returns users with that status
   - With pagination → Returns paginated results

3. Toggle Status:
   - Valid user ID → Success
   - Invalid user ID → Error 404
   - Non-admin user → Error 403

4. Delete User:
   - Valid user ID → Success (soft delete)
   - Invalid user ID → Error 404
   - Non-admin user → Error 403
   - Cannot delete self → Error 400

5. Get Stats:
   - Returns correct counts
   - Updates after user creation
   - Updates after user deletion
